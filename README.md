npm install  # Установить зависимости

npm run start  # Запустить бота

ALCHEMY_RPC_URL="https://solana-mainnet.g.alchemy.com/v2/токен_ALCHEMY"

DEXSCREENER_API_URL="https://api.dexscreener.com/latest/dex/tokens/"
